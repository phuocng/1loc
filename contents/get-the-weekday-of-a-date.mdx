title: Get the weekday of a date
updated: '2021-10-13'
---

**JavaScript version**

```js
// `date` is a Date object
const getWeekday = (date) =>
    ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'][date.getDay()];
```

**TypeScript version**

```js
const getWeekday = (date: Date): string =>
    ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'][date.getDay()];
```
